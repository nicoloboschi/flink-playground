#!/bin/bash

curl --output /tmp/avro-tools.jar https://dlcdn.apache.org/avro/avro-1.11.1/java/avro-tools-1.11.1.jar
export SCHEMA='{"type":"record","namespace":"com.example","name":"json","fields":[{ "name":"field1", "type": "string" }]}'
export AVROFILE=/tmp/avro-message.avro
java -jar /tmp/avro-tools.jar jsontofrag "$SCHEMA" use-cases/json-message2.json > $AVROFILE

echo "config use default
admin --admin-url http://localhost:8090 topics delete -f tavro
client produce -f /tmp/avro-message.avro -vs 'avro:{\"type\":\"record\",\"namespace\":\"com.example\",\"name\":\"json\",\"fields\":[{ \"name\":\"field1\", \"type\": \"string\" }]}' tavro
" | pulsar-shell - 

CREATE TABLE tavro (
    field1 STRING
) WITH (   'connector' = 'pulsar',   'topics' = 'public/default/tavro',   'service-url' = 'pulsar://localhost:6650',   
'admin-url' = 'http://localhost:8090', 'format' = 'avro' );


CREATE TABLE tavro (
    field1 STRING,
    field2 int,
    `event_time` TIMESTAMP(3) METADATA,
    `message_id` STRING METADATA,
    `publish_time` STRING METADATA,
    `properties` MAP<STRING, STRING> METADATA
) WITH (   'connector' = 'pulsar',   'topics' = 'public/default/tavro',   'service-url' = 'pulsar://localhost:6650',   
'admin-url' = 'http://localhost:8090', 'format' = 'avro' );

SELECT * FROM tavro;
